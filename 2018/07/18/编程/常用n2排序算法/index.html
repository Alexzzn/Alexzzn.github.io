// 冒泡排序
void sort1(int arr[],int n) {

int i,j = 0;
for (i = 0; i < n; ++i) {

    for (j = 0; j < n - i -1; ++j) {

        if (arr[j]>arr[j+1]){

            swap(arr[j],arr[j+1]);
        }
    }
}

}

// 选择排序
void sort2(int arr[],int n) {

for (int i = 0; i < n; ++i) {

    // 每次循环第一个i默认最小值
    int minIndex = i;
    for (int j = i+1; j < n; ++j) {

        if (arr[minIndex] > arr[j]) {
            minIndex = j;
        }
    }
    // 替换最小值
    swap(arr[minIndex],arr[i]);
}

}

// 插入排序

void sort3(int arr[], int n) {

for (int i = 1; i < n; ++i) {

    // 对比i前几个元素
    for (int j = i; j >= 1; j--) {

        if (arr[j]<arr[j-1]) {
            swap(arr[j],arr[j-1]);
        }
    }
}

}
